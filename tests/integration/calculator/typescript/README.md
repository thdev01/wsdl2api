  await client.someOperation(request);
} catch (error) {
  const apiError = error as APIError;
  console.error('Status:', apiError.status);
  console.error('Message:', apiError.message);
  if (apiError.fault) {
    console.error('SOAP Fault:', apiError.fault);
  }
}
